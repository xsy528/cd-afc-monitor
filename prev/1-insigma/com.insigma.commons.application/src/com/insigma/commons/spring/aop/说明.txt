三种方法的区别：
 1.TransactionInterceptor.java 
      拦截以Service结尾的方法的所有类的方法，捕捉所有异常重新抛ApplicationException
 2.ThrowAdvice.java
      只捕捉commit以及getTransaction方法所有抛的异常
 3.ExceptionAdvice.java
      捕捉service包以及子包下的所有方法抛的异常
 4.再配置上第三中方法比较麻烦，还需要引入一个外包，改spring文件头，效果差不多；
   <!-- Advice-around 拦截器实现自动代理拦截 -->
	<bean id="transactionInterceptor"
		class="com.insigma.spring.aop.TransactionInterceptor">
		
	</bean>
	<bean id="autoProxy"
		class="org.springframework.aop.framework.autoproxy.BeanNameAutoProxyCreator">
		<property name="beanNames">
			<list>
				<value>*Service</value>
			</list>
		</property>
		<property name="interceptorNames">
			<value>transactionInterceptor</value>
		</property>
	</bean>
	<bean id="autoProxyCreator" class="org.springframework.aop.framework.autoproxy.DefaultAdvisorAutoProxyCreator">
	</bean>
